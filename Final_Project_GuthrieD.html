# Helper function for lagging
lagm <- function(x, k = 1) {
  n <- nrow(x)
  pad <- matrix(NA, k, ncol(x))
  rbind(pad, x[1:(n - k), ])
}

# Generating 5 lags for each row
lags <- lapply(1:5, function(k) lagm(xdata, k))
names(lags) <- paste0("L", 1:5)
arframe <- do.call(cbind, lags)
arframe <- arframe[-(1:5), ]  # Remove first 5 rows

# Scaling response: log_volume, aligning with predictors
y <- scale(NYSE$log_volume)[-(1:5)]

# Matching train indicator
istrain <- istrain[-(1:5)]

# Cleaning NA rows
keep <- complete.cases(arframe)
arframe <- arframe[keep, ]
y <- y[keep]
istrain <- istrain[keep]

# Converting to 3D array 
n_timesteps <- 5
n_features <- ncol(xdata)

X <- array(as.numeric(unlist(arframe)), 
           dim = c(nrow(arframe), n_timesteps, n_features))

# Splitting into training and test sets
X_train <- X[istrain, , ]
X_test <- X[!istrain, , ]
y_train <- y[istrain]
y_test <- y[!istrain]
